Sub CheckPSA()
    'Aim: To generate iterations of age-dependent proportion based on random sampling of log-normal prameters.

    ' === Define worksheet and range ===
    Dim ws_check As Worksheet

    Set ws_check = ThisWorkbook.Sheets("Check")

    ' Define iteration number
    Dim iterations As Long
    iterations = 30 ' Set the number of iterations

    ' Record age-specific proportion through iterations
    Dim i As Long

    For i = 1 To iterations

        ' Status bar info
        Application.StatusBar = "Running iteration " & i & " of " & iterations

        ws_check.Cells(7, 7 + i).Value = i
        ws_check.Range(ws_check.Cells(8, 7 + i), ws_check.Cells(108, 7 + i)).Value = ws_check.Range("G8:G108").Value
    Next i

    ' Reset status bar
    Application.StatusBar = False
End Sub
